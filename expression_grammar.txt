EXPRESSION -> NOT IDENTIFIER
EXPRESSION -> NOT EXPRESSION
EXPRESSION -> IDENTIFIER
EXPRESSION -> EXPRESSION B
B -> BINARY_OP EXPRESSION
EXPRESSION -> LP C
C -> EXPRESSION RP
