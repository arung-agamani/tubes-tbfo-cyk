WITH -> WITHWORD WITH0
WITH0 -> IDENTIFIER WITH1
WITH1 -> LP WITH2
WITH2 -> IDENTIFIER WITH3
WITH3 -> RP WITH4
WITH4 -> COLON
WITH4 -> AS COLON
WITHWORD -> 'WITH'

CLASS -> CLASSWORD CLASS1
CLASS1 -> IDENTIFIER COLON
CLASSWORD -> 'CLASS'

DEF -> DEFWORD DEF1
DEF1 -> IDENTIFIER DEF2
DEF2 -> LP DEF3
DEF3 -> DEF4
DEF3 -> IDENTIFIER DEF4
DEF3 -> STRING DEF4
DEF3 -> NUMBER DEF4
DEF3 -> DEF3 DEFINBETWEEN
DEFINBETWEEN -> COMA DEF3
DEF4 -> RP COLON
DEFWORD -> 'DEF'

WHILE_METHOD -> WHILE W1
W1 -> IDENTIFIER W4
W1 -> EXPRESSION W4
W1 -> LP W2
W2 -> IDENTIFIER W3
W2 -> EXPRESSION W3
W3 -> RP W4
W4 -> COLON NEWLINE
W4 -> COLON
WHILE -> 'WHILE'
NEWLINE -> 'NEWLINE'

EXPRESSION -> IDENTIFIER EXP1
EXP1 -> COMPARE_OP IDENTIFIER
EXPRESSION -> NOT IDENTIFIER
EXPRESSION -> NOT EXPRESSION
EXPRESSION -> IDENTIFIER
EXPRESSION -> EXPRESSION EXP2
EXP2 -> BINARY_OP EXPRESSION
EXPRESSION -> LP EXP3
EXP3 -> EXPRESSION RP
COMPARE_OP -> 'DOUBLEEQUAL'
COMPARE_OP -> 'GREATER_OR_EQUAL_THAN'
COMPARE_OP -> 'LESS_OR_EQUAL_THAN'
COMPARE_OP -> 'GREATER_THAN'
COMPARE_OP -> 'LESS_THAN'
BINARY_OP -> 'AND'
BINARY_OP -> 'OR'
NOT -> 'NOT'
IDENTIFIER -> 'IDENTIFIER'
LP -> 'LP'
RP -> 'RP'
